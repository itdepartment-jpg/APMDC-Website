<script setup lang="ts">
import { ref,computed } from 'vue';
import { Head, router, useForm } from '@inertiajs/vue3';
import AuthenticatedLayout from '@/Layouts/AuthenticatedLayout.vue';

import {
    DocumentIcon,
    PencilIcon,
    TrashIcon,
    PlusIcon,
    ArrowDownTrayIcon,
    XMarkIcon,
    ChevronLeftIcon,
    ChevronRightIcon
} from '@heroicons/vue/24/outline';
import Modal from '@/Components/Modal.vue';
import InputLabel from '@/Components/InputLabel.vue';
import TextInput from '@/Components/TextInput.vue';
import TextArea from '@/Components/TextArea.vue';
import PrimaryButton from '@/Components/PrimaryButton.vue';
import SecondaryButton from '@/Components/SecondaryButton.vue';
import DangerButton from '@/Components/DangerButton.vue';
import Pagination from '@/Components/Pagination.vue';
import { useToast } from '@/Composables/useToast';

interface PressRelease {
    id: number;
    title: string;
    summary: string;
    content: string;
    released_at: string;
    pdf_path: string | null;
    pdf_url: string | null;
}

interface Props {
    pressReleases: {
        data: PressRelease[];
        current_page: number;
        last_page: number;
        links: any[];
    };
    canEdit: boolean;
}

const props = defineProps<Props>();
const { showSuccess, showError } = useToast();

// State management
const showFormModal = ref(false);
const showDeleteModal = ref(false);
const releaseToDelete = ref<PressRelease | null>(null);
const releaseToEdit = ref<PressRelease | null>(null);
const isLoading = ref(false);

// Form handling
const form = useForm({
    title: '',
    summary: '',
    content: '',
    released_at: new Date().toISOString().slice(0, 16),
    pdf: null as File | null,
    pdf_path: null as string | null
});

// Computed properties
const modalTitle = computed(() =>
    releaseToEdit.value ? 'Edit Press Release' : 'Create New Press Release'
);

const formattedDate = (dateString: string) => {
    return new Date(dateString).toLocaleDateString('en-US', {
        year: 'numeric',
        month: 'short',
        day: 'numeric',
    });
};

// Methods
const openFormModal = (release: PressRelease | null = null) => {
    releaseToEdit.value = release;

    if (release) {
        form.title = release.title;
        form.summary = release.summary;
        form.content = release.content;
        form.released_at = release.released_at.slice(0, 16);
        form.pdf_path = release.pdf_path;
    } else {
        form.reset();
    }

    showFormModal.value = true;
};

const closeFormModal = () => {
    showFormModal.value = false;
    form.reset();
    releaseToEdit.value = null;
};

const openDeleteModal = (release: PressRelease) => {
    releaseToDelete.value = release;
    showDeleteModal.value = true;
};

const closeDeleteModal = () => {
    showDeleteModal.value = false;
    releaseToDelete.value = null;
};

const handlePdfChange = (e: Event) => {
    const file = (e.target as HTMLInputElement)?.files?.[0];
    if (file) {
        form.pdf = file;
    }
};

const removePdf = () => {
    form.pdf = null;
    form.pdf_path = null;
};

const submitForm = async () => {
    try {
        isLoading.value = true;

        const method = releaseToEdit.value ? 'put' : 'post';
        const url = releaseToEdit.value
            ? route('press-releases.update', releaseToEdit.value.id)
            : route('press-releases.store');

        await form[method](url, {
            preserveScroll: true,
            onSuccess: () => {
                showSuccess(releaseToEdit.value
                    ? 'Press release updated successfully'
                    : 'Press release created successfully');
                closeFormModal();
            },
            onError: (errors) => {
                showError('There was an error processing your request');
            },
        });
    } finally {
        isLoading.value = false;
    }
};

const deleteRelease = async () => {
    if (!releaseToDelete.value) return;

    try {
        isLoading.value = true;
        await router.delete(route('press-releases.destroy', releaseToDelete.value.id), {
            preserveScroll: true,
            onSuccess: () => {
                showSuccess('Press release deleted successfully');
                closeDeleteModal();
            },
            onError: () => {
                showError('Failed to delete press release');
            },
        });
    } finally {
        isLoading.value = false;
    }
};

const loadPage = (url: string) => {
    router.visit(url, {
        preserveState: true,
        preserveScroll: true,
        onSuccess: () => {
            window.scrollTo({ top: 0, behavior: 'smooth' });
        },
    });
};
</script>

<template>
    <Head title="Press Releases Management" />

    <AuthenticatedLayout>
        <template #header>
            <h2 class="text-2xl font-bold leading-tight text-gray-800">
                Press Releases Management
            </h2>
        </template>

        <div class="py-6">
            <div class="max-w-7xl mx-auto sm:px-6 lg:px-8">
                <div class="bg-white overflow-hidden shadow-sm sm:rounded-lg">
                    <div class="p-6">
                        <!-- Header with create button -->
                        <div class="flex flex-col sm:flex-row justify-between items-start sm:items-center mb-6 gap-4">
                            <div>
                                <h1 class="text-2xl font-bold text-gray-900">Press Releases</h1>
                                <p class="text-sm text-gray-500 mt-1">
                                    Manage all press releases and official statements
                                </p>
                            </div>
                            <PrimaryButton
                                v-if="canEdit"
                                @click="openFormModal()"
                                class="flex items-center gap-2"
                            >
                                <PlusIcon class="h-5 w-5" />
                                <span>New Press Release</span>
                            </PrimaryButton>
                        </div>

                        <!-- Empty state -->
                        <div v-if="pressReleases.data.length === 0" class="border-2 border-dashed border-gray-200 rounded-lg p-8 text-center">
                            <div class="flex flex-col items-center justify-center">
                                <DocumentIcon class="h-12 w-12 text-gray-400 mb-4" />
                                <h3 class="text-lg font-medium text-gray-900 mb-1">
                                    No press releases yet
                                </h3>
                                <p class="text-sm text-gray-500 mb-4">
                                    Get started by creating your first press release
                                </p>
                                <PrimaryButton
                                    v-if="canEdit"
                                    @click="openFormModal()"
                                    class="flex items-center gap-2"
                                >
                                    <PlusIcon class="h-5 w-5" />
                                    <span>New Press Release</span>
                                </PrimaryButton>
                            </div>
                        </div>

                        <!-- Press Releases table -->
                        <div v-else class="overflow-x-auto">
                            <table class="min-w-full divide-y divide-gray-200">
                                <thead class="bg-gray-50">
                                <tr>
                                    <th scope="col" class="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">
                                        Title
                                    </th>
                                    <th scope="col" class="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">
                                        Summary
                                    </th>
                                    <th scope="col" class="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">
                                        Release Date
                                    </th>
                                    <th scope="col" class="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">
                                        PDF
                                    </th>
                                    <th v-if="canEdit" scope="col" class="px-6 py-3 text-right text-xs font-medium text-gray-500 uppercase tracking-wider">
                                        Actions
                                    </th>
                                </tr>
                                </thead>
                                <tbody class="bg-white divide-y divide-gray-200">
                                <tr v-for="release in pressReleases.data" :key="release.id">
                                    <td class="px-6 py-4 whitespace-nowrap">
                                        <div class="text-sm font-medium text-gray-900">
                                            {{ release.title }}
                                        </div>
                                    </td>
                                    <td class="px-6 py-4">
                                        <div class="text-sm text-gray-500 line-clamp-2">
                                            {{ release.summary }}
                                        </div>
                                    </td>
                                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500">
                                        {{ formattedDate(release.released_at) }}
                                    </td>
                                    <td class="px-6 py-4 whitespace-nowrap">
                                        <a
                                            v-if="release.pdf_url"
                                            :href="release.pdf_url"
                                            target="_blank"
                                            class="inline-flex items-center text-blue-600 hover:text-blue-800"
                                        >
                                            <DocumentIcon class="h-5 w-5 mr-1" />
                                            Download
                                        </a>
                                        <span v-else class="text-sm text-gray-400">None</span>
                                    </td>
                                    <td v-if="canEdit" class="px-6 py-4 whitespace-nowrap text-right text-sm font-medium">
                                        <div class="flex justify-end space-x-2">
                                            <button
                                                @click="openFormModal(release)"
                                                class="text-blue-600 hover:text-blue-900 p-1 rounded-md hover:bg-blue-50"
                                                title="Edit"
                                            >
                                                <PencilIcon class="h-5 w-5" />
                                            </button>
                                            <button
                                                @click="openDeleteModal(release)"
                                                class="text-red-600 hover:text-red-900 p-1 rounded-md hover:bg-red-50"
                                                title="Delete"
                                            >
                                                <TrashIcon class="h-5 w-5" />
                                            </button>
                                        </div>
                                    </td>
                                </tr>
                                </tbody>
                            </table>

                            <!-- Pagination -->
                            <Pagination
                                :links="pressReleases.links"
                                :current-page="pressReleases.current_page"
                                :last-page="pressReleases.last_page"
                                @page-changed="loadPage"
                                class="mt-6"
                            />
                        </div>
                    </div>
                </div>
            </div>
        </div>

        <!-- Create/Edit Modal -->
        <Modal :show="showFormModal" @close="closeFormModal" max-width="3xl">
            <div class="p-6">
                <div class="flex items-center justify-between mb-4">
                    <h3 class="text-lg font-medium text-gray-900">
                        {{ modalTitle }}
                    </h3>
                    <button
                        @click="closeFormModal"
                        class="text-gray-400 hover:text-gray-500 focus:outline-none"
                    >
                        <XMarkIcon class="h-6 w-6" />
                    </button>
                </div>

                <form @submit.prevent="submitForm">
                    <div class="space-y-6">
                        <!-- Title -->
                        <div>
                            <InputLabel for="title" value="Title" required />
                            <TextInput
                                id="title"
                                v-model="form.title"
                                type="text"
                                class="mt-1 block w-full"
                                :class="{ 'border-red-500': form.errors.title }"
                                required
                            />
                            <p v-if="form.errors.title" class="mt-1 text-sm text-red-600">
                                {{ form.errors.title }}
                            </p>
                        </div>

                        <!-- Summary -->
                        <div>
                            <InputLabel for="summary" value="Summary" required />
                            <TextArea
                                id="summary"
                                v-model="form.summary"
                                rows="3"
                                class="mt-1 block w-full"
                                :class="{ 'border-red-500': form.errors.summary }"
                                required
                                maxlength="500"
                            />
                            <p class="mt-1 text-xs text-gray-500">
                                {{ form.summary.length }}/500 characters
                            </p>
                            <p v-if="form.errors.summary" class="mt-1 text-sm text-red-600">
                                {{ form.errors.summary }}
                            </p>
                        </div>

                        <!-- Content -->
                        <div>
                            <InputLabel for="content" value="Content" required />
                            <TextArea
                                id="content"
                                v-model="form.content"
                                rows="6"
                                class="mt-1 block w-full"
                                :class="{ 'border-red-500': form.errors.content }"
                                required
                            />
                            <p v-if="form.errors.content" class="mt-1 text-sm text-red-600">
                                {{ form.errors.content }}
                            </p>
                        </div>

                        <div class="grid grid-cols-1 md:grid-cols-2 gap-6">
                            <!-- Release Date -->
                            <div>
                                <InputLabel for="released_at" value="Release Date" required />
                                <TextInput
                                    id="released_at"
                                    v-model="form.released_at"
                                    type="datetime-local"
                                    class="mt-1 block w-full"
                                    :class="{ 'border-red-500': form.errors.released_at }"
                                    required
                                />
                                <p v-if="form.errors.released_at" class="mt-1 text-sm text-red-600">
                                    {{ form.errors.released_at }}
                                </p>
                            </div>

                            <!-- PDF Upload -->
                            <div>
                                <InputLabel for="pdf" value="PDF Attachment" />
                                <div class="mt-1">
                                    <div v-if="form.pdf_path || form.pdf" class="flex items-center gap-3 mb-2">
                                        <DocumentIcon class="h-8 w-8 text-gray-400" />
                                        <span class="text-sm text-gray-600">
                      {{ form.pdf?.name || 'Existing PDF attached' }}
                    </span>
                                        <button
                                            type="button"
                                            @click="removePdf"
                                            class="text-red-500 hover:text-red-700"
                                        >
                                            <XMarkIcon class="h-5 w-5" />
                                        </button>
                                    </div>
                                    <input
                                        id="pdf"
                                        type="file"
                                        accept=".pdf"
                                        @change="handlePdfChange"
                                        class="block w-full text-sm text-gray-500 file:mr-4 file:py-2 file:px-4 file:rounded-md file:border-0 file:text-sm file:font-semibold file:bg-blue-50 file:text-blue-700 hover:file:bg-blue-100"
                                    />
                                    <p class="mt-1 text-sm text-gray-500">
                                        PDF only (Max 5MB)
                                    </p>
                                    <p v-if="form.errors.pdf" class="mt-1 text-sm text-red-600">
                                        {{ form.errors.pdf }}
                                    </p>
                                </div>
                            </div>
                        </div>
                    </div>

                    <!-- Form Actions -->
                    <div class="mt-6 flex justify-end space-x-3">
                        <SecondaryButton @click="closeFormModal" type="button">
                            Cancel
                        </SecondaryButton>
                        <PrimaryButton type="submit" :disabled="form.processing">
                            {{ releaseToEdit ? 'Update' : 'Create' }} Press Release
                        </PrimaryButton>
                    </div>
                </form>
            </div>
        </Modal>

        <!-- Delete Confirmation Modal -->
        <Modal :show="showDeleteModal" @close="closeDeleteModal" max-width="md">
            <div class="p-6">
                <div class="flex items-center justify-between mb-4">
                    <h3 class="text-lg font-medium text-gray-900">
                        Delete Press Release
                    </h3>
                    <button
                        @click="closeDeleteModal"
                        class="text-gray-400 hover:text-gray-500 focus:outline-none"
                    >
                        <XMarkIcon class="h-6 w-6" />
                    </button>
                </div>

                <div class="flex items-start">
                    <div class="mr-3 flex-shrink-0">
                        <ExclamationCircleIcon class="h-12 w-12 text-red-500" />
                    </div>
                    <div>
                        <h4 class="text-lg font-medium text-gray-900 mb-2">
                            Are you sure you want to delete this press release?
                        </h4>
                        <p class="text-sm text-gray-500">
                            This action cannot be undone. All data associated with this press release will be permanently removed.
                        </p>
                        <p v-if="releaseToDelete" class="mt-2 font-medium">
                            "{{ releaseToDelete.title }}"
                        </p>
                    </div>
                </div>

                <div class="mt-6 flex justify-end space-x-3">
                    <SecondaryButton @click="closeDeleteModal" type="button">
                        Cancel
                    </SecondaryButton>
                    <DangerButton
                        @click="deleteRelease"
                        type="button"
                        :disabled="isLoading"
                    >
                        Delete Press Release
                    </DangerButton>
                </div>
            </div>
        </Modal>
    </AuthenticatedLayout>
</template>
